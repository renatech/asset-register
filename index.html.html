Google Sheets Backup

How it works

This app saves data in your browser automatically. Google Sheets acts as a cloud backup — every time you add, edit or delete an asset, it syncs a copy to a Google Sheet you control. No server required.

Setup (one-time, ~3 minutes)

Go to sheets.google.com and create a new spreadsheet called VF Asset Register.
Click Extensions → Apps Script. Delete any code there and paste the script below, then click Save (floppy disk icon).
Click Deploy → New deployment → Web app. Set "Who has access" to Anyone. Click Deploy and copy the Web App URL.
Paste that URL in the field below and click Save. Done — all changes will sync automatically.
function doPost(e){ var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Assets') || ss.insertSheet('Assets'); sheet.clearContents(); var headers = ['ID','Name','Category','Asset No','Serial No','Condition', 'Acquisition','Unit Cost','Quantity','Sub-Total','Acquired', 'Assigned To','Date Assigned','Date Returned','Notes']; sheet.appendRow(headers); data.forEach(function(r){ sheet.appendRow([r._id,r.name,r.category,r.asset_number||'', r.serial||'',r.condition,r.acquisition||'',r.unit_cost||0, r.quantity||1,r.sub_total||0,r.acquired||'',r.staff||'', r.date_assigned||'',r.date_returned||'',r.notes||'' ]); }); return ContentService.createTextOutput('OK'); } function doGet(e){ var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Assets'); if(!sheet) return ContentService.createTextOutput(JSON.stringify([])); var rows = sheet.getDataRange().getValues(); var headers = rows[0]; var data = rows.slice(1).map(function(r){ var obj={}; headers.forEach(function(h,i){obj[h]=r[i];}); return obj; }); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
