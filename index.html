Firebase rules — Set your database to restricted mode. In Firebase Console → Realtime Database → Rules, paste the rules below and publish.

Replace the above with these stricter rules once you have Firebase connected — restricts write access and validates data structure. For production, generate a secret key and add auth validation.

Firebase — Real-time multi-user backend

Connect Firebase so all staff share the same live data. Changes appear instantly on every device. Free tier is more than enough for this register.

Step 1
Go to console.firebase.google.com → Create project → VF Asset Register
Step 2
Click Realtime Database → Create database → Start in test mode → Enable
Step 3
Copy your database URL: https://vf-asset-register-xxx.firebaseio.com
Step 4
Paste URL below → Connect. All data syncs instantly across all devices.

Google Sheets Backup

Go to sheets.google.com → create spreadsheet named VF Asset Register.
Click Extensions → Apps Script. Paste script below → Save.
Click Deploy → New deployment → Web app. Access: Anyone. Copy the URL.
Paste URL below → Save & Test.
function doPost(e){ var data=JSON.parse(e.postData.contents); var ss=SpreadsheetApp.getActiveSpreadsheet(); var sheet=ss.getSheetByName('Assets')||ss.insertSheet('Assets'); sheet.clearContents(); sheet.appendRow(['ID','Name','Category','Asset No','Serial','Condition', 'Source','Donor','Unit Cost','Qty','Sub-Total','Acquired', 'Current Value','Assigned To','Date Assigned','Date Returned','Notes']); data.forEach(function(r){ sheet.appendRow([r._id,r.name,r.category,r.asset_number||'',r.serial||'', r.condition,r.acquisition||'',r.donor||'',r.unit_cost||0,r.quantity||1, r.sub_total||0,r.acquired||'',r.dep_value||0,r.staff||'', r.date_assigned||'',r.date_returned||'',r.notes||'']); }); return ContentService.createTextOutput('OK'); } function doGet(){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sheet=ss.getSheetByName('Assets'); if(!sheet)return ContentService.createTextOutput(JSON.stringify([])); var rows=sheet.getDataRange().getValues();var h=rows[0]; return ContentService.createTextOutput(JSON.stringify( rows.slice(1).map(function(r){var o={};h.forEach(function(k,i){o[k]=r[i];});return o;}) )).setMimeType(ContentService.MimeType.JSON); }
